018    package org.apache.commons.configuration.web;
019    
020    import java.util.Collections;
021    import java.util.Enumeration;
022    import java.util.Iterator;
023    
024    import javax.servlet.FilterConfig;
025    
026    /**
027     * A configuration wrapper around a {@link FilterConfig}. This configuration is
028     * read only, adding or removing a property will throw an
029     * UnsupportedOperationException.
030     *
031     * @author <a href="mailto:ebourg@apache.org">Emmanuel Bourg</a>
032     * @version $Id: ServletFilterConfiguration.java 1211130 2011-12-06 20:57:19Z oheger $
033     * @since 1.1
034     */
035    public class ServletFilterConfiguration extends BaseWebConfiguration
036    {
037        /** Stores the wrapped filter config.*/
038        protected FilterConfig config;
039    
040        /**
041         * Create a ServletFilterConfiguration using the filter initialization parameters.
042         *
043         * @param config the filter configuration
044         */
045        public ServletFilterConfiguration(FilterConfig config)
046        {
047            this.config = config;
048        }
049    
050        public Object getProperty(String key)
051        {
052            return handleDelimiters(config.getInitParameter(key));
053        }
054    
055        public Iterator<String> getKeys()
056        {
057            // According to the documentation of getInitParameterNames() the
058            // enumeration is of type String.
059            @SuppressWarnings("unchecked")
060            Enumeration<String> en = config.getInitParameterNames();
061            return Collections.list(en).iterator();
062        }
063    }
