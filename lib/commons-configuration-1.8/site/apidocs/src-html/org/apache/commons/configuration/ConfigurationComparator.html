018    package org.apache.commons.configuration;
019    
020    /**
021     * Comparator for configurations interface.
022     *
023     * @since 1.0
024     *
025     * @author <a href="mailto:herve.quiroz@esil.univ-mrs.fr">Herve Quiroz</a>
026     * @version $Id: ConfigurationComparator.java 1208783 2011-11-30 21:12:45Z oheger $
027     */
028    public interface ConfigurationComparator
029    {
030        /**
031         * Compare two configuration objects.
032         *
033         * @param a the first configuration
034         * @param b the second configuration
035         * @return true if the two configurations are identical according to
036         *         the implemented rules
037         */
038        boolean compare(Configuration a, Configuration b);
039    }
