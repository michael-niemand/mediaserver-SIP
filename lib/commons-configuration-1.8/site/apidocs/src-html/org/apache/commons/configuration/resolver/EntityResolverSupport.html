017    package org.apache.commons.configuration.resolver;
018    
019    import org.xml.sax.EntityResolver;
020    
021    /**
022     * Interface that identifies the class as using an EntityResolver
023     * @author <a href="http://commons.apache.org/configuration/team-list.html">Commons
024     * Configuration team</a>
025     * @since 1.7
026     * @version $Id: EntityResolverSupport.java 1206761 2011-11-27 16:35:34Z oheger $
027     */
028    public interface EntityResolverSupport
029    {
030        /**
031         * Return the EntityResolver associated with the class.
032         * @return The EntityResolver.
033         */
034        EntityResolver getEntityResolver();
035    
036        /**
037         * Set the EntityResolver to associate with this class.
038         * @param resolver The EntityResolver
039         */
040        void setEntityResolver(EntityResolver resolver);
041    }
